pipeline {
    agent any
    environment{
	APP_DIR='/var/lib/jenkins/workspace/order-ms/spring'
	JAR_FILE='order-ms-0.0.1-SNAPSHOT.jar'
    IMAGE = 'aafrozbanu/order-ms:latest'
    CREDENTIALS_ID = 'docker-registry-login-id'
	}
    stages {
         stage('Clean workspace'){
            steps{
               cleanWs() 
            }
        }
        stage('Cloning Git Repo') {
            steps {
                script{
                    sh 'git clone "https://github.com/afrozbanugit/spring.git"'
                }
            }
        }
        stage('Build Docker Image') {
            steps {
                script {
                    sh 'docker build -t ${IMAGE}'
                }
            }
        }
        stage('Push Docker Image') {
            steps {
                script {
                    docker.withRegistry('https://hub.docker.com/', "${CREDENTIALS_ID}") {
                        docker.image("${IMAGE}").push()
                    }
                }
            }
        }
    }
}
